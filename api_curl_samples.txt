# API cURL Samples
# Replace http://localhost with your actual server URL

# 1. Upload File
# Content must be Base64 encoded string of the file.
# Example content: "SGVsbG8gV29ybGQ=" (Hello World)
curl -X POST http://localhost/api/upload \
     -H "Content-Type: application/json" \
     -d '{"filename": "example.txt", "content": "SGVsbG8gV29ybGQ="}'

# 2. Receive File
# Replace "1234" with the actual PIN received from the upload step.
curl -X POST http://localhost/api/receive \
     -H "Content-Type: application/json" \
     -d '{"pin": "1234"}'

# 3. Share Text
# Content should be Base64 encoded.
# Example content: "U2VjcmV0IE1lc3NhZ2U=" (Secret Message)
curl -X POST http://localhost/api/share-text \
     -H "Content-Type: application/json" \
     -d '{"content": "U2VjcmV0IE1lc3NhZ2U="}'

# 4. View Text
# Replace "5678" with the actual PIN received from the share text step.
curl -X POST http://localhost/api/view-text \
     -H "Content-Type: application/json" \
     -d '{"pin": "5678"}'
